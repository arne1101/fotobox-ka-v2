1. Autostart und Start-/Stopp Script

- sudo nano /etc/init.d/fotobox

#!/bin/bash

### BEGIN INIT INFO
# Provides:          fotobox
# Required-Start:    $remote_fs $syslog
# Required-Stop:     $remote_fs $syslog
# Default-Start:     2 3 4 5
# Default-Stop:      0 1 6
# Short-Description: Fotobox initscript
# Description:       This service is used to manage the fotobox
### END INIT INFO


case "$1" in 
    start)
        echo "Starting fotobox"
        perl /var/www/FotoboxApp/bin/app.pl>/dev/null &
	starman /var/www/FotoboxApp/bin/app.pl>/dev/null &
        ;;
    stop)
        echo "Stopping fotobox"
        killall perl
        ;;
    restart)
       echo "Restarting fotobox"
       killall perl
       perl /var/www/FotoboxApp/bin/app.pl>/dev/null &
       starman /var/www/FotoboxApp/bin/app.pl>/dev/null &
       ;;
    *)
        echo "Usage: /etc/init.d/servod start|stop|restart"
        exit 1
        ;;
esac

exit 0

- sudo chmod +x /etc/init.d/fotobox

- sudo update-rc.d fotobox defaults
