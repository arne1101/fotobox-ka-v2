	2. gphoto2
	
		sudo apt-get install gphoto2

		
        http://www.geekingabout.com/controlling-a-nikon-d5200-with-raspberry-pi-and-gphoto2-2-5-2.html

		Install dependencies
		
		apt-get install -y libltdl-dev libusb-dev libexif-dev libpopt-dev
		Create a temp folder to work in
		mkdir gphoto2-temp-folder
		cd gphoto2-temp-folder
		
		Download, compile and install libusb
		
		wget http://ftp.de.debian.org/debian/pool/main/libu/libusbx/libusbx_1.0.11.orig.tar.bz2
		tar xjvf libusbx_1.0.17.orig.tar.bz2
		cd libusbx-1.0.11/
		sudo ./configure
		sudo make
		sudo make install
		cd ..

		a) Download, compile and install libgphoto2 2.5.5.1
		
		wget http://sourceforge.net/projects/gphoto/files/libgphoto/2.5.5.1/libgphoto2-2.5.5.1.tar.bz2
		tar xjf libgphoto2-2.5.5.1.tar.bz2
		cd libgphoto2-2.5.5.1
		sudo ./configure
		sudo make
		sudo make install
		cd ..
		
		b) Download, compile and install gphoto2 2.5.5

		wget http://downloads.sourceforge.net/project/gphoto/gphoto/2.5.5/			gphoto2-2.5.5.tar.gz
		tar xzvf gphoto2-2.5.5.tar.gz
		cd gphoto2-2.5.5
		sudo ./configure
		sudo make
		sudo make install
		cd ..

		c) sudo ldconfig

	4. ftp server installieren
	
		sudo apt-get install vsftpd
		
		sudo nano /etc/vsftpd.conf
		
		Suche nach den folgenden Einträgen und ändere diese ab. 
		Achtung Schreibweise mit YES und NO und dem Kommentarzeichen #. 
		
		Anonymous_enable=NO
		Local_enable=YES
		Write_enable=YES
		Ascii_upload_enable=YES
		Ascii_download_enable=YES
		
		/etc/init.d/vsftpd restart
